Example of Using Custom Directive
HostListenner: First you need to import HostListenner from @angular/core HostListenner able us to subscribe to events from the DOM hosting this element. ElementRef: an object include reference to the DOM element. [appInputFormat]="'uppercase'" work in case of one attribute pass to the directive. the name of the Input property is the directive selector. 

  // for example
  export class InputFormatDirective {

  @Input('appInputFormat') format;
  
  constructor(private el: ElementRef) {
  }

  @HostListener('focus') onfocus() {
    console.log('on Focus');
  }

  @HostListener('blur') onblur() {
    let value: string = this.el.nativeElement.value;
    if (this.format == 'lowercase') {
      this.el.nativeElement.value = value.toLowerCase();
    }
    else {
      this.el.nativeElement.value = value.toLocaleUpperCase();
    }
  }
}

// the html is like:
<input type="text" [appInputFormat]="'uppercase'">
