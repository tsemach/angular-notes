Getiting Data: read posts from server
  1. reading posts from server (array of jsons)
  2. display each with *ngFor

// code: reading posts from server
posts: any[];
url = 'http://localhost:3000/http/';

constructor(private http: Http) {
  http.get(this.url + 'get')
    .subscribe(response =>  {
        console.log(response.json());
        this.posts = response.json();
      }
    );
  }

// html: display all posts with 
<ul class="list-group">
  <li 
    *ngFor=let post of posts"
    class="list-group-item">
    { { post.title } }
  </li>
</ul>
  • {{ post.title }}
