Update post: update a post using http put request
  1. Get input value from input element
  2. Send a post reguest to server
  3. Server reply with Id of the new adding item
  4. Update a post

// send put to server - url/:id, post
updatePost(post) {
  this.http.put(this.url + 'put/' + post.id, post)
    .subscribe(response => {
      console.log(response);
      this.input = JSON.stringify(response.json());
    });
  }

// html: input new title and display the post
<input
  [(ngModel)]="input"
  (keyup.enter)="createPost(title)" #title
  type="text" 
  class="form-control">

<ul class="list-group">
  <li 
    *ngFor="let post of posts"
    <button
      (click)="updatePost(post)"

    </button>
